-----------------version actualizada de las tablas--------------

CREATE TABLE sala(
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    ubicacion TEXT NOT NULL,
    descripcion TEXT NOT NULL
);

CREATE TABLE tipo_interactivo (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    descripcion TEXT NOT NULL
);

CREATE TABLE status_interactivo (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL
);

CREATE TABLE interactivo (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    numero_serie VARCHAR(100) NOT NULL,
    modelo_computadora VARCHAR(100),
    marca_computadora VARCHAR(100),
    fecha_instalado DATE NOT NULL,
    sala_id INTEGER NOT NULL REFERENCES sala(id) ON DELETE CASCADE,
    tipo_id INTEGER NOT NULL REFERENCES tipo_interactivo(id) ON DELETE CASCADE,
    status_id INTEGER NOT NULL REFERENCES status_interactivo(id) ON DELETE CASCADE
);

CREATE TABLE tipo_mantenimiento (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    descripcion TEXT NOT NULL
);

CREATE TABLE dep_area (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    descripcion TEXT NOT NULL
);

CREATE TABLE status_mantenimiento (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL
);

CREATE TABLE nivel_prioridad (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL UNIQUE,
    descripcion TEXT,
    nivel SMALLINT NOT NULL UNIQUE
);

CREATE TABLE reporte_falla (
    id SERIAL PRIMARY KEY,
    interactivo_id INTEGER NOT NULL REFERENCES interactivo(id) ON DELETE CASCADE,
    descripcion TEXT NOT NULL,
    foto VARCHAR(255),
    fecha_reporte DATE NOT NULL,
    estado_id INTEGER NOT NULL REFERENCES status_mantenimiento(id) ON DELETE RESTRICT,
    prioridad_id INTEGER REFERENCES nivel_prioridad(id) ON DELETE RESTRICT
);

CREATE TABLE tarea_mantenimiento (
    id SERIAL PRIMARY KEY,
    interactivo_id INTEGER NOT NULL REFERENCES interactivo(id) ON DELETE CASCADE,
    reporte_id INTEGER NOT NULL REFERENCES reporte_falla(id) ON DELETE CASCADE,
    area_id INTEGER NOT NULL REFERENCES dep_area(id) ON DELETE CASCADE,
    tipo_id INTEGER NOT NULL REFERENCES tipo_mantenimiento(id) ON DELETE CASCADE,
    titulo VARCHAR(100) NOT NULL,
    descripcion TEXT NOT NULL,
    fecha_programada DATE NOT NULL,
    fecha_realizada DATE,
    estado_id INTEGER NOT NULL REFERENCES status_mantenimiento(id) ON DELETE RESTRICT
);


CREATE TABLE reporte_mantenimiento (
    id SERIAL PRIMARY KEY,
    interactivo_id INTEGER NOT NULL REFERENCES interactivo(id) ON DELETE CASCADE,
    tarea_id INTEGER NOT NULL REFERENCES tarea_mantenimiento(id) ON DELETE CASCADE,
    observaciones TEXT NOT NULL,
    fecha DATE NOT NULL,
    estado_id INTEGER NOT NULL REFERENCES status_mantenimiento(id) ON DELETE RESTRICT
);

CREATE TABLE estado_tarea (
    id SERIAL PRIMARY KEY,
    tarea_id INTEGER NOT NULL REFERENCES tarea_mantenimiento(id) ON DELETE CASCADE,
    estado_anterior VARCHAR(50) NOT NULL,
    estado_nuevo VARCHAR(50) NOT NULL,
    fecha_cambio DATE NOT NULL,
    status_id INTEGER NOT NULL REFERENCES status_mantenimiento(id) ON DELETE RESTRICT DEFAULT 1
);

CREATE TABLE usuario (
    id SERIAL PRIMARY KEY,
    nombre_usuario VARCHAR(100) NOT NULL UNIQUE,
    correo VARCHAR(150) NOT NULL,
    contrasena_hash TEXT NOT NULL,
    rol VARCHAR(50) NOT NULL,
    fecha_creacion TIMESTAMP WITHOUT TIME ZONE NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE perfil (
    id SERIAL PRIMARY KEY,
    usuario_id INTEGER NOT NULL UNIQUE REFERENCES usuario(id) ON DELETE CASCADE,
    nombre_completo VARCHAR(150) NOT NULL,
    telefono VARCHAR(20) NOT NULL,
    foto_url TEXT,
    direccion TEXT NOT NULL,
    fecha_nacimiento DATE NOT NULL,
    genero VARCHAR(20) NOT NULL
);

CREATE TABLE proyector (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    numero_serie VARCHAR(100) NOT NULL UNIQUE,
    modelo VARCHAR(100) NOT NULL,
    marca VARCHAR(100) NOT NULL,
    fecha_instalado DATE NOT NULL,
    sala_id INTEGER NOT NULL REFERENCES sala(id) ON DELETE CASCADE,
    status_id INTEGER NOT NULL REFERENCES status_interactivo(id) ON DELETE CASCADE
);

CREATE TABLE calendarizacion (
    id SERIAL PRIMARY KEY,
    titulo VARCHAR(150) NOT NULL,
    descripcion TEXT,
    fecha_inicio TIMESTAMP WITHOUT TIME ZONE NOT NULL,
    fecha_fin TIMESTAMP WITHOUT TIME ZONE NOT NULL,
    interactivo_id INTEGER REFERENCES interactivo(id) ON DELETE CASCADE,
    tarea_id INTEGER REFERENCES tarea_mantenimiento(id) ON DELETE CASCADE,
    creado_por_id INTEGER REFERENCES usuario(id) ON DELETE SET NULL,
    creado_el TIMESTAMP WITHOUT TIME ZONE NOT NULL DEFAULT CURRENT_TIMESTAMP
);
